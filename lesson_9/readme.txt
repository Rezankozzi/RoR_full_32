В этой папке лежит задание к уроку "Метапрограммирование"

Папка <<modules>> с подключаемыми модулями: instance_counter.rb (счетчик количества экземпляров класса);
                                            manufacturer.rb (данные о производителе); 
                                            test_module.rb (создание нескольких объектов для тестирования работы
                                                            программы);
                                            accessors.rb (динамическое добавление атрибутов класса)
                                            validation.rb (модуль валидации для объектов классов)


Файлы с описанием классов: train.rb; train_cargo.rb, train_passenger.rb,
                           wagon.rb, wagon_cargo.rb, wagon_passenger.rb
                           station.rb;
                           route.rb;

Папка <<interfaces>> с описанием интерфейса пользователя: interface.rb
                                и подключаемыми модулями: data_interface.rb;
                                                          train_interface.rb;
                                                          station_interface.rb;
                                                          route_interface.rb;
                                                          wagon_interface.rb

Файл запуска программы: main.rb
